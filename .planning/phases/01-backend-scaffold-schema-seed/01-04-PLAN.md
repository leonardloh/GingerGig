    ```python
    """Alembic env — async, importing Settings (D-16) and app.models for autogenerate."""
    import asyncio
    from logging.config import fileConfig

    from alembic import context
    from sqlalchemy import pool
    from sqlalchemy.engine import Connection
    from sqlalchemy.ext.asyncio import async_engine_from_config

    # Importing app.models registers all 11 tables on Base.metadata
    import app.models  # noqa: F401
    from app.core.config import settings
    from app.db.base import Base

    # Alembic Config object
    config = context.config

    # Override the placeholder DSN with the live setting (D-16)
    config.set_main_option("sqlalchemy.url", settings.database_url)

    if config.config_file_name is not None:
        fileConfig(config.config_file_name)

    target_metadata = Base.metadata


    def run_migrations_offline() -> None:
        url = config.get_main_option("sqlalchemy.url")
        context.configure(
            url=url,
            target_metadata=target_metadata,
            literal_binds=True,
            dialect_opts={"paramstyle": "named"},
            compare_type=True,
            compare_server_default=True,
        )
        with context.begin_transaction():
            context.run_migrations()


    def do_run_migrations(connection: Connection) -> None:
        context.configure(
            connection=connection,
            target_metadata=target_metadata,
            compare_type=True,
            compare_server_default=True,
        )
        with context.begin_transaction():
            context.run_migrations()


    async def run_async_migrations() -> None:
        connectable = async_engine_from_config(
            config.get_section(config.config_ini_section, {}),
            prefix="sqlalchemy.",
            poolclass=pool.NullPool,
        )
        async with connectable.connect() as connection:
            await connection.run_sync(do_run_migrations)
        await connectable.dispose()


    def run_migrations_online() -> None:
        asyncio.run(run_async_migrations())


    if context.is_offline_mode():
        run_migrations_offline()
    else:
        run_migrations_online()
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Migration file (committed source) → ApsaraDB schema | Migrations run with elevated DDL privileges; bad SQL = data loss |
| `gingergig_test` database → `gingergig` (production) | Migration script is the same artifact; if it works on test, it ships to prod (Phase 8) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | T (Tampering) | An autogenerated migration silently drops a column on the next phase's autogenerate | mitigate | `compare_type=True` and `compare_server_default=True` in env.py force autogenerate to flag type drifts; every migration must be hand-reviewed before commit. The plan explicitly mandates "hand-review the generated file" in step 3. |
| T-04-02 | I (Information Disclosure) | Hardcoded ApsaraDB DSN with credentials in `alembic.ini` committed to git | mitigate | `alembic.ini`'s `sqlalchemy.url` is left as a placeholder; env.py overrides with `settings.database_url` from `.env` (D-16). The verify step `grep -E '^sqlalchemy.url\s*=\s*postgresql'` would catch a hardcoded production DSN. |
| T-04-03 | E (EoP) | Phase 2+ developer running `alembic upgrade head` against `gingergig` (production) by accident, applying an untested migration | accept | Mitigated by the `[BLOCKING]` test against `gingergig_test` first. Phase 8 deploy will document a separate ApsaraDB role with restricted DDL. |
| T-04-04 | D (DoS) | Migration acquires an ACCESS EXCLUSIVE lock on production tables long enough to block traffic | accept | Phase 1 is greenfield (CREATE TABLE only) — no concurrent traffic exists. Future migrations should use `CONCURRENTLY` for index creation; documented in the migration `script.py.mako` template (Phase 8 hardens). |
| T-04-05 | T | `alembic_version` table being deleted/clobbered, leaving the DB in an unknown migration state | accept | Standard alembic risk; documented recovery path in README ("alembic downgrade base && alembic upgrade head && make seed"). |
