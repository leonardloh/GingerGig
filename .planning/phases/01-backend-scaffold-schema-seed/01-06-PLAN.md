    ```python
    """Pytest fixtures for Phase 1 (D-17, D-18, D-19).

    - Session-scoped engine pointing at gingergig_test (TEST_DATABASE_URL).
    - Per-test SAVEPOINT rollback so tests don't poison each other.
    - httpx.AsyncClient via ASGITransport — supported async-end-to-end since httpx 0.28.
    """
    from __future__ import annotations

    import os
    from collections.abc import AsyncIterator

    import httpx
    import pytest
    import pytest_asyncio
    from httpx import ASGITransport
    from sqlalchemy.ext.asyncio import AsyncEngine, AsyncSession, async_sessionmaker

    from app.core.config import Settings, settings
    from app.db.session import build_engine
    from app.deps.db import get_db
    from app.main import app


    def _test_settings() -> Settings:
        """Build a Settings instance forced to the test DSN regardless of .env."""
        test_url = os.environ.get("TEST_DATABASE_URL")
        if not test_url:
            raise RuntimeError(
                "TEST_DATABASE_URL is not set; tests require gingergig_test on ApsaraDB."
            )
        return settings.model_copy(update={"database_url": test_url})


    @pytest_asyncio.fixture(scope="session")
    async def engine() -> AsyncIterator[AsyncEngine]:
        """Session-scoped engine pointing at gingergig_test."""
        eng = build_engine(_test_settings())
        # Also wire the FastAPI app to the same engine for client tests:
        app.state.engine = eng
        yield eng
        await eng.dispose()


    @pytest_asyncio.fixture
    async def db_session(engine: AsyncEngine) -> AsyncIterator[AsyncSession]:
        """Per-test SAVEPOINT rollback (D-18)."""
        async with engine.connect() as conn:
            outer_trans = await conn.begin()
            sm = async_sessionmaker(bind=conn, expire_on_commit=False)
            async with sm() as session:
                # Begin nested SAVEPOINT so the test can commit at the session level
                # without breaking the outer rollback.
                await session.begin_nested()
                try:
                    yield session
                finally:
                    await outer_trans.rollback()


    @pytest_asyncio.fixture
    async def client(db_session: AsyncSession, engine: AsyncEngine) -> AsyncIterator[httpx.AsyncClient]:
        """ASGITransport client (D-19) wired to the SAVEPOINT-isolated session."""

        async def _override_get_db() -> AsyncIterator[AsyncSession]:
            yield db_session

        app.dependency_overrides[get_db] = _override_get_db
        try:
            transport = ASGITransport(app=app)
            async with httpx.AsyncClient(transport=transport, base_url="http://test") as ac:
                yield ac
        finally:
            app.dependency_overrides.clear()
    ```

    Note: the `engine` fixture sets `app.state.engine = eng` directly because the FastAPI lifespan does NOT run when the app is invoked through ASGITransport without a startup event. This shortcut is acceptable for Phase 1 — the engine is the only state lifespan would have set.

    ```python
    """D-17 #3: scripts/seed.py is idempotent across two consecutive runs.

    Row counts unchanged AND siti's password_hash unchanged (D-08 preservation).
    """
    import os

    import pytest
    from sqlalchemy import text
    from sqlalchemy.ext.asyncio import AsyncEngine


    TABLES = [
        "users", "companion_links", "listings", "listing_menu_items",
        "bookings", "reviews", "companion_alerts",
        "companion_alert_preferences", "timeline_events",
    ]


    async def _snapshot(engine: AsyncEngine) -> tuple[dict[str, int], str]:
        async with engine.connect() as conn:
            counts = {}
            for t in TABLES:
                counts[t] = (await conn.execute(text(f"SELECT COUNT(*) FROM {t}"))).scalar_one()
            siti_hash = (
                await conn.execute(
                    text("SELECT password_hash FROM users WHERE email='siti@gingergig.my'")
                )
            ).scalar_one_or_none()
        return counts, siti_hash


    @pytest.mark.asyncio
    async def test_seed_idempotent(engine: AsyncEngine) -> None:
        """The seed has been applied at least once by Plan 05's [BLOCKING] task.
        Re-run it once more inline and verify state is unchanged."""
        # First snapshot (after Plan 05's seed)
        counts1, hash1 = await _snapshot(engine)
        assert counts1["users"] >= 8, f"expected at least 8 users (3 demo + 5 providers), got {counts1['users']}"
        assert hash1 is not None, "siti@gingergig.my missing — Plan 05 didn't seed?"

        # Re-run seed
        os.environ["ALLOW_SEED"] = "1"
        os.environ["DATABASE_URL"] = os.environ["TEST_DATABASE_URL"]
        from scripts import seed
        rc = await seed.main()
        assert rc == 0, f"seed exit code {rc}"

        # Second snapshot
        counts2, hash2 = await _snapshot(engine)
        assert counts1 == counts2, f"row counts drifted: {counts1} -> {counts2}"
        assert hash1 == hash2, "siti's password_hash changed across re-run (D-08 violation)"


    @pytest.mark.asyncio
    async def test_demo_accounts_password_demo(engine: AsyncEngine) -> None:
        """All 3 DEMO_ACCOUNTS bcrypt-verify with password "demo" (DATA-07)."""
        import bcrypt
        async with engine.connect() as conn:
            for email in ("siti@gingergig.my", "amir@gingergig.my", "faiz@gingergig.my"):
                h = (
                    await conn.execute(
                        text(f"SELECT password_hash FROM users WHERE email='{email}'")
                    )
                ).scalar_one_or_none()
                assert h is not None, f"{email} not seeded"
                assert bcrypt.checkpw(b"demo", h.encode("ascii")), (
                    f"{email} password_hash does not verify against 'demo'"
                )
    ```

    ```python
    """Recommended: enforce that none of the forbidden libraries appear anywhere
    in app/ or scripts/. Verifies PITFALLS.md compliance at the source-code level.
    """
    import pathlib
    import re

    BACKEND = pathlib.Path(__file__).parent.parent
    FORBIDDEN_PATTERNS = [
        r"^\s*(?:import|from)\s+passlib",
        r"^\s*(?:import|from)\s+jose",
        r"^\s*(?:import|from)\s+aioredis",
        r"^\s*(?:import|from)\s+oss2",
        r"^\s*(?:import|from)\s+psycopg2",
    ]


    def test_no_forbidden_imports() -> None:
        offenders: list[str] = []
        for path in (BACKEND / "app", BACKEND / "scripts"):
            for py in path.rglob("*.py"):
                src = py.read_text()
                for pattern in FORBIDDEN_PATTERNS:
                    if re.search(pattern, src, re.MULTILINE):
                        offenders.append(f"{py}: matches /{pattern}/")
        assert not offenders, "Forbidden imports found:\n" + "\n".join(offenders)
    ```
